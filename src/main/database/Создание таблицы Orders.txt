CREATE TABLE orders
(id BIGINT IDENTITY,
description VARCHAR(500),
clientID BIGINT NOT NULL,
dataofcreation DATE NOT NULL,
dateofcompletion DATE check (dateofcompletion >= dataofcreation) NOT NULL,
price FLOAT NOT NULL,
status VARCHAR(50) check (status in ('Запланирован','Выполнен','Принят клиентом')) NOT NULL,
PRIMARY KEY (id),
FOREIGN KEY (clientID) REFERENCES client(id))
